#!/usr/bin/python3.8

# pylint: disable=line-too-long,missing-function-docstring,missing-module-docstring

import os
import sys
import logging

NAAM = os.path.splitext(os.path.basename(__file__))[0]
BOM_UTF8 = b'\xef\xbb\xbf'
NEWFILEBYTES = [123, 3, 255, 0, 100]

def main():

    print(sys.version_info)
    logging.basicConfig(filename='{}.log'.format(NAAM), filemode='w', level="INFO", format='%(asctime)s - %(levelname)s - %(message)s', datefmt='%d-%b-%y %H:%M:%S')
    logging.debug('This is a debug message')
    logging.info('This is an info message')
    logging.warning('This is a warning message')
    logging.error('This is an error message')
    logging.critical('This is a critical message')

    with open('{0}.txt'.format(NAAM), 'wb') as fp1:
        fp1.write(BOM_UTF8)
        fp1.write(bytearray(NEWFILEBYTES))

if __name__ == '__main__':
    main()
